var isPalindrome = function (head) {
  let cur = head
  const values = []
  while (cur) {
    values.push(cur.val)
    cur = cur.next
  }

  let left = 0
  let right = values.length - 1
  while (left <= right) {
    if (values[left] !== values[right]) {
      return false
    }
    left++
    right--
  }
  return true
};
