var reverseList = function (head) {
  if (!head) return head
  let prev = null
  let cur = head
  let next = null
  while (cur) {
    next = cur.next // 这里要注意在更改 cur.next 之前需要缓存 cur.next
    cur.next = prev
    prev = cur
    cur = next
  }

  return prev
};

// 思路：逐个遍历改变节点的next指向，需要注意在更改 cur.next 之前需要缓存 cur.next, 代码如下 next = cur.next
